def plusMinus(arr):
    # Write your code here
    n = len(arr)  
    p = 0  
    ne = 0  
    z = 0  

    for i in range(n): 
        if arr[i] > 0:
            p += 1
        elif arr[i] < 0:
            ne += 1
        else:
            z += 1

    
    
    print(round(p / n, 6))
    print(round(ne / n, 6))
    print(round(z / n, 6))
